CREATE DATABASE universe;

\c universe

CREATE TABLE galaxy(galaxy_id SERIAL PRIMARY KEY, name VARCHAR(30) NOT NULL UNIQUE, num_stars INT NOT NULL, diameter_in_tril_miles NUMERIC(9), num_galaxies_within_5_mil_light_years INT, is_spiral BOOLEAN, has_black_hole BOOLEAN);

CREATE TABLE star(star_id SERIAL PRIMARY KEY, name VARCHAR(30) NOT NULL UNIQUE, num_planets INT NOT NULL, diameter_in_tril_miles NUMERIC(9), num_stars_within_10_light_years INT, CONSTRAINT fk_galaxy FOREIGN KEY galaxy_id REFERENCES galaxy(galaxy_id) ON DELETE SET NULL);

CREATE TABLE planet(planet_id SERIAL PRIMARY KEY, name VARCHAR(30) NOT NULL UNIQUE, num_moons INT NOT NULL, diameter_in_miles NUMERIC(9), FOREIGN KEY star_id REFERENCES star(star_id) ON DELETE SET NULL);

CREATE TABLE moon(moon_id SERIAL PRIMARY KEY, name VARCHAR(30) NOT NULL UNIQUE, orbit_time_in_hours INT NOT NULL, diameter_in_miles NUMERIC(9), FOREIGN KEY planet_id REFERENCES planet(planet_id) ON DELETE SET NULL);
